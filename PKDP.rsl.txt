// This product uses the environmental data downloaded by GR2Analyst to
// determine the maximum KDP between the height of the melting layer 
// and 1 km below it for values greater than 1 deg/km as a proxy for KDP 
// cores. Additionally, it filters out KDP values below the melting
// layer if the KDP values above the melting layer are 90% or greater than
// that of the KDP values below the melting layer to remove data from KDP
// columns 
//
// Written by: Kristofer S. Tuftedal, Ph.D. (CIWRO/NSSL)

Product
 {
  Name = "PKDP"
  Category = "KDP"
  resolution = high
 }

Program
{
  float kdp;         // Variable to hold the current value of KDP during iteration
  float hgt_ml;      // Variable to hold the height of the melting layer
  float hgt_neg20;
  float hgt_bottom;  // Variable to hold the height 1 km below the melting layer
  float max_kdp;     // Variable to hold the maximum KDP value
  float zh;
  float cc;
  float zdr;

  hgt_ml = HGTfromTEMP(0); // Get the height of the melting layer (0°C)
  hgt_bottom = hgt_ml - 1000; // Calculate the height 1 km below the melting layer

  max_kdp = constants.ND; // Initialize max_kdp with No Data value

  // Enumerate through the heights from hgt_bottom to hgt_ml
  enumerate levels from hgt_bottom to hgt_ml
  {
    kdp = KDPfromHGT(level.height); // Get the KDP value at the current level height
	zh = BRfromHGT(level.height);
	cc = CCfromHGT(level.height);
	zdr = ZDRfromHGT(level.height);
    if (kdp >= 1.0 && zh >= 20 && cc >= 0.8 && zdr >= 0 && zdr <= 5) { // Only consider KDP values >= 1 degree/km where meteorological echoes are present
      if (max_kdp == constants.ND || kdp > max_kdp) {
        max_kdp = kdp; // Update max_kdp if the current KDP value is greater
      }
    }
  }
  
  // Remove data if KDP values above the melting layer meet or exceed 90% of that below it
  hgt_neg20 = HGTfromTEMP(-20);
  enumerate levels from hgt_neg20
  {
    kdp = KDPfromHGT(level.height); // Get the KDP value at the current level height
	zh = BRfromHGT(level.height);
	zdr = ZDRfromHGT(level.height);
    if (kdp >= 1.0 && zh >=20 && zdr > 0) { // Only consider KDP values >= 1 degree/km
      if (max_kdp == constants.ND || kdp >= (max_kdp - (max_kdp*0.1))) {
        max_kdp = constants.ND; // Update max_kdp if the current KDP value is greater
      }
    }
  }

  return max_kdp; // Return the maximum KDP value found in the range
}
